\section*{\vspace{-1em} \textendash{} Protocol \textendash{}}
\markright{Protocol}

\subsection*{Day 1 \textendash{} Sample preparation}

\alert{In each step, samples can be rocked on a nutator or on an orbital (horizontal) shaker.}

\begin{enumerate}[series = steps]
	\item Dissect the desired adult tissue in 100\% methanol.
	\item Wash samples for \underline{\quarter} each with a graded series of \fivehunmicrol{} (or \onemil) of \textbf{methanol (\qtylist{60;30}{\percent}) in PBSTx} at RT.
	\item \step{wash}{\quarter}{\fivehunmicrol{} (or \onemil)}{PBSTx}{RT}
	\item \textit{(optional)} \step{rock}{\qty{2}{\minute}}{\fivehunmicrol{} (or \onemil)}{\qty{2.6}{\ug\per\ml} proteinase K}{RT}
	\item \textit{(optional)} \step{wash}{two times without incubation}{\fivehunmicrol{} (or \onemil)}{PBST}{RT}
	\item \textit{(optional)} \step{post-fix}{\halfhour}{\fivehunmicrol{} (or \onemil)}{\qty{3.7}{\percent} PFA}{RT}
	\item \textit{(optional)} \step{wash}{\qtytimes{2}{10}{\minute}}{\fivehunmicrol{} (or \onemil)}{PBSTx}{RT}
	\item \step{incubate}{\qty{2}{\hour}}{\fivehunmicrol{} (or \onemil)}{PBSTx}{RT}
	\item \step{incubate}{\qty{> 12}{\hour} (overnight)}{\fivehunmicrol{} (or \onemil)}{a blocking solution of \qty{6}{\percent} goat serum in PBSTx}{RT}
\end{enumerate}

\subsection*{Day 2 \textendash{} Primary antibody reaction}
\begin{enumerate}[series = steps]
	\item \step{incubate}{\qtyrange{24}{48}{\hour}}{\fivehunmicrol}{1:40 primary antibody in \qty{6}{\percent} goat serum solution}{\fourdegree}
\end{enumerate}

\subsection*{Day 3 \textendash{} Primary antibody reaction}
\begin{enumerate}[series = steps]
	\item \textit{ongoing incubation with primary antibody solution}
\end{enumerate}

\subsection*{Day 4 \textendash{} Secondary antibody reaction}
\begin{enumerate}[series = steps]
	\item \step{wash}{\qtytimes{6}{30}{\minute}}{\fivehunmicrol (or \onemil)}{TBSTx}{RT}
	\item \textit{optional} \step{incubate}{\qty{1}{\hour}}{\fivehunmicrol{} (or \onemil)}{a blocking solution of \qty{6}{\percent} goat serum in PBSTx}{RT}
	\item \step{incubate}{\qty{> 12}{\hour} (overnight)}{\fivehunmicrol}{1:20 secondary antibody in \qty{6}{\percent} goat serum}{RT}
\end{enumerate}

\subsection*{Day 5 \textendash{} IF conclusion}

\begin{enumerate}[series = steps]
	\item \step{wash}{\qtytimes{2}{5}{\minute}, \qtytimes{5}{15}{\minute}, and \qtytimes{1}{5}{\minute}}{\fivehunmicrol{} (or \onemil)}{PBSTx}{RT}
	\item \step{incubate}{\qtyrange{30}{60}{\minute}}{\onemil}{\qty{50}{\percent} glycerol in \pbs}{RT}
	\item \step{incubate}{\qtyrange{30}{60}{\minute}}{\onemil}{\qty{75}{\percent} glycerol in \pbs}{RT}
	\item Collect samples from the \qty{75}{\percent} glycerol solution and place them on a cleaned (bridged, if necessary) slide.
	\item Add \qtyrange{\sim 15}{20}{\ul} of mounting medium (or VECTASHIELD\textregistered{} PLUS Antifade Mounting Medium with DAPI [H-2000]) directly on samples.
\end{enumerate}

\bigskip\alert{Adjust the amount of mounting medium according to the cover glass dimensions, the quantity of samples, and the thickness of the bridge.}

\begin{enumerate}[resume = steps]
	\item	Seal the slide with nail polish and store in the dark at \fourdegree.
\end{enumerate}
