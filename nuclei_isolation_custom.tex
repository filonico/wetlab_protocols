\subsection*{Procedure}

\alert{Make sure to set the temperature of the \qty{4}{\degreeCelsius} centrifuge beforehand.}
\alert{Always \textbf{work on ice}, unless otherwise specified.}
\alert{Always keep the working environment and materials as much \textbf{RNase-free} as possible.}
\alert{Place all the working solutions, tubes, and homogenizers on ice before starting.}
\alert{Keep the frozen samples at \qty{-80}{\degreeCelsius} until use. Otherwise, place them in dry ice.}

\begin{enumerate}[series = steps]
	\item Add \textbf{\qtyrange{2}{5}{\ml} of ice-cold working lysis buffer} into a Dounce homogenizer.
	\item Transfer the  samples from the tube to the homogenizers using tweezers.
\end{enumerate}

\bigskip\alert{If the starting material is too much, place it on a dry-ice-cooled petri dish, and split it in multiple samples using a sterile scaple blade.}

\begin{enumerate}[resume = steps]
	\item Homogenize samples on ice, with a maximum of 10 loose-pestel strokes, and a maximum of 10 tight-pestle strokes.
	\item Rinse the pestles back into the homogenizers with \textbf{\qtyrange{100}{500}{\ul} of lysis buffer}.
\end{enumerate}

\bigskip\alert{If the sample does not fully dissociat, incubate on ice with the lysis buffer for a couple of minutes.}

\begin{enumerate}[resume = steps]
	\item Filter the homogenate through a \qty{100}{\um} cell strainer into a \qty{15}{\ml} Falcon tube, using \qty[scientific-notation = false, group-digits = false]{1000}{\ul} low retention tips.
	\item Rinse the cell strainer with \textbf{\pbs{} for single cell}.
	\item Filter the homogenate through a \qty{40}{\um} Flowmi\textregistered{} cell strainer into a clean a \qty{15}{\ml} Falcon tube, using \qty[scientific-notation = false, group-digits = false]{1000}{\ul} low retention tips.
	\item \label{wash1} Spin samples at \qtyrange{500}{900}{\g} for \qty{5}{\minute} at \fourdegree.
	\item Discard the supernatant.
	\item \label{wash3} Resuspend the nuclei with \textbf{\qty{2}{\ml} of working nuclei wash buffer}, gently pipetting with \qty[scientific-notation = false, group-digits = false]{1000}{\ul} low retention tips.
	\item Repeat steps \ref{wash1}\textendash\ref{wash3} twice.
	\item Resuspend samples in \textbf{\qty{500}{\ul} of \pbs{} for the single-nuclei workflow}.
\end{enumerate}

\begin{enumerate}
	\item Add Tris-\ce{HCl} (pH \num{7.5}), \ce{NaCl}, and \ce{MgCl2} to combin;
	\item add IGEPAL\textregistered{} CA-630 and BSA, mixing gently to avoid foaming;
	\item add nuclease-free water to reach a final volume of \qty{10}{\ml};
	\item filter sterilise with a \qty{0.2}{\um} filter;
	\item store at \qty{4}{\degreeCelsius} for 3--4 weeks, or at \qty{-20}{\degreeCelsius} for up to 6 months.
\end{enumerate}

\bigskip\alert{Avoid repeated freeze-thaw cycles. Thaw on ice. Mix gently to avoid foaming.}
