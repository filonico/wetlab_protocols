\subsection*{Procedure}

\alert{Make sure to set the temperature of the \qty{4}{\degreeCelsius} centrifuge beforehand.}
\alert{Always \textbf{work on ice}, unless otherwise specified.}
\alert{Always keep the working environment and materials as much \textbf{RNase-free} as possible.}
\alert{Place all the working solutions, tubes, and homogenizers on ice before starting.}
\alert{Keep the frozen samples at \qty{-80}{\degreeCelsius} until use. Otherwise, place them in dry ice.}

\begin{enumerate}[series = steps]
	\item Add \textbf{\qtyrange{2}{5}{\ml} of ice-cold working lysis buffer} into a Dounce homogenizer; keep in ice.
	\item Place the sample onto the lid of a \qty{5}{\ml} eppendorf tube filled with \qtyrange[scientific-notation = false, group-digits = false]{500}{1000}{\ul} of pre-chilled lysis buffer.
	\item Using a surgical sterile scalpel blade, \textbf{cut the sample into small chunks}.
	\item Transfer the  samples from the tube to the homogenizers using tweezers and/or a pipette, depending on chunk sizes.
	\item \textbf{Homogenize samples on ice}, with a maximum of 10 loose-pestel strokes, and a maximum of 10 tight-pestle strokes.
	\item Let big pieces of tissues to settle at the bottom of the homogenizer.
	\item Repeat homogenization if the tissue does not look dissociated.
\end{enumerate}

\bigskip\alert{Always rinse the pestles back into the homogenizers with \textbf{\qtyrange{100}{500}{\ul} of lysis buffer}.}
	
\begin{enumerate}[resume = steps]
	\item \textbf{Filter the homogenate} through a \qtyrange{30}{40}{\um} cell strainer into a \qty{5}{\ml} eppendorf tube, using \qty[scientific-notation = false, group-digits = false]{1000}{\ul} low retention tips.
	\item Rinse the cell strainer with nuclei lysis buffer.
	\item \textbf{Spin samples} at \qty{500}{\g} for \qty{10}{\minute} at \fourdegree.
	\item Discard the supernatant and leave \qty{~50}{\ul} of buffer.
	\item Add \textbf{\qty{600}{\ul} of nuclei wash buffer} drop-wise without disturbing the pellet.
	\item \textbf{Resuspend the nuclei} by gently pipetting with a \qty[scientific-notation = false, group-digits = false]{200}{\ul} low retention tip, making sure they stay close to the bottom of the tube.
	\item \textbf{Spin samples} at \qty{500}{\g} for \qty{10}{\minute} at \fourdegree.
	\item Discard the supernatant and leave \qty{~25}{\ul} of buffer.
	\item Add \textbf{\qty{50}{\ul} of wash buffer} and resuspend the pellet by gently pipetting.
	\item Possibly filter the nuclei suspension through a \qty{10}{\um} cell strainer to avoid aggregates.
\end{enumerate}

\begin{enumerate}
	\item Add Tris-\ce{HCl} (pH \num{7.5}), \ce{NaCl}, and \ce{MgCl2} to combin;
	\item add IGEPAL\textregistered{} CA-630 and BSA, mixing gently to avoid foaming;
	\item add RNAse inhibitor;
	\item add \qty{4}{\ml} of nuclease-free water;
	\item add the protease inhibitor tablet and let dissolve, also by gently pipetting;
	\item add the remaining amount of nuclease-free water;
	\item filter sterilise with a \qty{0.2}{\um} filter;
	\item use same day.
\end{enumerate}
