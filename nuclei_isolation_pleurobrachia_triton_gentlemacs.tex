\subsection*{Procedure}

\alert{Make sure to set the temperature of the \qty{4}{\degreeCelsius} centrifuge beforehand.}
\alert{Make sure to place the gentleMACS\textregistered{} machine in the cold room beforehand.}
\alert{Always \textbf{work on ice}, unless otherwise specified.}
\alert{Always keep the working environment and materials as much \textbf{RNase-free} as possible.}
\alert{Place all the working solutions, tubes, and homogenizers on ice before starting.}
\alert{Keep the frozen samples at \qty{-80}{\degreeCelsius} until use. Otherwise, place them in dry ice.}

\begin{enumerate}[series = steps]
	\item Add \textbf{\qty{4}{\ml} of homogenization buffer} into a gentleMACS\textregistered{} C-tube; keep in ice.
	\item Move materials ands yourslef to the cold room.
	\item Cut samples into small pieces, but try to not thaw them: work on a Petri dish placed on dry ice and use a sterile scaple with forceps.
	\item Transfer the chopped samples into the C-tube and place on the gentleMACS\textregistered{} machine.
	\item Homogenize samples with 5 cycles of the program \singlecurlyquotes{m\_brain\_03.01} (runs for \qty{1}{\minute}).
\end{enumerate}

\bigskip\alert{This program is available on the gentleMACS\textregistered{} machine v1.02. Newer machines have programs specific for nuclei isolation.}
	
\begin{enumerate}[resume = steps]
	\item Incubate homogenized samples on ice for \qty{5}{\minute}.
	\item Filter the homogenized samples through a \qty{30}{\um} MACS\textregistered{} cell strainer into a \qty{15}{\ml} low-bind falcon tube; use a wide-boar \qty{1000}{\ul} pipette tip.
	\item Wash the MACS\textregistered{} C-tube and cell strainer with leftover homogenization buffer.
	\item Centrifuge sample at \qty{500}{\g} for \qty{10}{\minute} at \fourdegree.
	\item Remove and discard the supernatant.
\end{enumerate}

\bigskip\alert{At this stage, if having multiple tubes with nuclei originating from the same sample, or if samples do not yield enough nuclei, the suspensions can be merged into a single new falcon tube.}
	
\begin{enumerate}[resume = steps]
	\item Resuspend samples in \textbf{\qty{1.5}{\ml} of nuclei wash/suspension buffer} (change this buffer according to the desired single-nuclei protocol).
	\item Centrifuge sample at \qty{500}{\g} for \qty{10}{\minute} at \fourdegree.
	\item Remove and discard the supernatant.
	\item If needed, filter samples through a \qty{10}{\um} pluriSelect pluriStrainer Mini cell strainer into a \qty{15}{\ml} low-bind falcon tube.
\end{enumerate}

\bigskip\alert{\qty{10}{\um} cell strainer may result in significant sample loss, as they get clogged very easily: ensure that the suspension is clean enough. If the cell strainer gets clogged, try to spin the tube with the strainer on top in the \fourdegree{} centrifuge.}

\begin{enumerate}[resume = steps]
	\item Remove and discard the supernatant, leaving behind \qty{~25}{\ul} of wash/suspension in the tube.
	\item Add \qty{15}{\ul} of nuclei wash/suspension buffer to reach a final volume of \qty{~40}{\ul}.
	\item Check nuclei quality and number using an automated counter and an AO/PI staining (1:1 dilution).
	\item Proceed directly to the single-cell capture protocol.
\end{enumerate}

\bigskip\alert{The suspension final volume should be adjusted according to the sample yield, in order to get a final nuclei concentration of \qty{~5000}{cells\per\ul}}.

\begin{enumerate}
	\item Thoroughly dissolve 1 tablet of protease inhibitor in \qty{1}{\ml} of nuclease-free water, by vigorously vortexin in a \qty{1.5}{\ml} eppendorf tube;
	\item add \qty{0.3}{\ml} of Triton X-100 to the protease inhibitor solution (dissolve the Triton X-100 by gentle pipetting to avoid bubbling);
	\item keep the protease inhibitor solution on ice;
	\item add \qty{2.5}{\ml} of \qty{1}{\molar} sucrose to a \qty{15}{\ml} falcon tube;
	\item add \qty{0.05}{\ml} of \qty{1}{\molar} \ce{MgCl2};
	\item add \qty{0.25}{\ml} of \qty{1}{\molar} \ce{KCl};
	\item add \qty{0.1}{\ml} of \qty{1}{\molar} Tris-\ce{HCl} (pH \num{8.0});
	\item add \qty{0.1}{\ml} of \qty{0.1}{\molar} DTT;
	\item add \qty{0.2}{\ml} of RNAse inhibitor;
	\item if available, add spermine to a final concentration of \qty{1}{\milli\molar};
	\item gently add the protease solution to the falcon tube;
	\item fill up to \qty{10}{\ml} by adding the remaining nuclease-free water;
	\item filter with a \qty{0.2}{\um} syringe filter.
	\item keep the working solution on ice.	
\end{enumerate}
